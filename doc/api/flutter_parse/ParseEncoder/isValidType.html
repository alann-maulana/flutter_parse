bool isValidType(dynamic value) {
  return value == null ||
      value is String ||
      value is num ||
      value is bool ||
      value is DateTime ||
      value is List ||
      value is Map ||
      value is Uint8List ||
      value is ParseObject ||
      value is ParseGeoPoint;
}
